\subsection*{The Topologically Enriched Category $\J$}\label{sec:J}\todo{this definition not worthwhile entire subsec}

We begin by defining the topologically enriched category $\J$, which is the indexing category of chain complexes in the topologically enriched setting.

\begin{defn}
We define the pointed topologically enriched category $\J$ as follows:
\begin{itemize}
    \item The objects of $\J$ are the integers together with a distinguished object $\infty$:
    \begin{equation*}
        \Ob(\J) = \mathbb{Z} \cup \{ \infty \}.
    \end{equation*}
    \item The morphism spaces $\Hom_{\J}(j, i)$ are defined for $i, j \in \mathbb{Z} \cup \{ \infty \}$ by:
    \begin{equation*}
        \Hom_{\J}(j, i) = 
        \begin{cases}
            \{ \id, \infty \} & \text{if } i = j, \\
            \{ d_i, \infty \} & \text{if } j = i + 1, \\
            (\Hlf^{J_{j,i}})_+ & \text{if } j > i + 1, \\
            \{ \infty \} & \text{if } i = \infty \text{ or } j = \infty,
        \end{cases}
    \end{equation*}
    where $X_+$ denotes the one-point compactification of a space $X$, and $\Hlf$ is the half-line $[0, \infty)$. The set $J_{j,i}$ is defined by:
    \begin{equation}\label{eq:J_indices}
        J_{j,i} = \{ \ell \in \mathbb{Z} \mid j > \ell > i \}.
    \end{equation}
    The spaces $\Hom_{\J}(j, i)$ are pointed at the point $\infty$.
    \item The map
    \begin{equation*}
        J_{k,j}\sqcup J_{j,i} \hookrightarrow J_{k,i}
    \end{equation*}
    induce a map
    \begin{equation*}
        \Hlf^{J_{k,j}}\times \Hlf^{J_{j,i}} \hookrightarrow \Hlf^{J_{k,i}}.
    \end{equation*}
    By taking $(-)_+$ we get the composition in $\J$:
    \begin{equation*}
        (\Hlf^{J_{k,j}})_+\smashSpaces (\Hlf^{J_{j,i}})_+ \to (\Hlf^{J_{k,i}})_+.
    \end{equation*}
\end{itemize}
\end{defn}

\begin{rem}
Intuitively, the morphisms in $\J$ correspond to the differentials and higher compositions in a chain complex. The space $\Hlf^{J_{j,i}}$ models the (higher) nullhomotopies between compositions of differentials.
\end{rem}

We call a functor $\Fun_+(\J,\Top_+)$ a \textbf{topological chain complex}.

\subsection{Topological chain complexes}
Our goal is to start with an embedded framed flow category and produce a topological chain complex. The idea is to use \cref{eq:ThomMap} in a compatible way to "collapse" the flow category and then to use $\J$ to encode maps between spheres and the higher homotopies.

\begin{constr}\label{con:topChOfCat}
    Let $\fC$ be a flow category with a neat embedding $e$ and normal framing $\varphi$.
    Assume that $\fC$ have bounded index $b \geq \norm{-} \geq a$, without loss of generality we have $a=0$. We define a pointed topologically enriched functor $K_{\fC} \in \Fun_+(\J,\Top_+)$ in the following way.
    
    First
    \begin{equation*}
        K_{\fC}(i) = \WedgeSpaces_{\norm{x}=i} S^{Nb}.
    \end{equation*}
    To define the morphisms, we use that $\Hom,\smashSpaces$ adjunction and get that we need to define maps:
    \begin{equation*}
        \Hom_{\J}(j,i)\smashSpaces \WedgeSpaces_{\norm{y}=j} S^{Nb} \to \WedgeSpaces_{\norm{x}=i} S^{Nb},
    \end{equation*}
    hence it's suffice to provide maps:
    \begin{equation*}
        g_{y,x}:\Hom_{\J}(j,i)\smashSpaces S^{Nb} \to S^{Nb}.
    \end{equation*}
    For each $y,x \in \fC$  with indices $j,i$.

    Let $y,x$ be as above, we define
    \begin{equation*}
        S^{N(\norm{y}-\norm{x})}\smashSpaces (\Hlf^{J_{y,x}})_+ \to  Th(e_{y,x}) \to S^{N(\norm{y}-\norm{x})}.
    \end{equation*}
    where the first map is the collapse map of $e_{y,x}$ and the second is (the compactification of) the projection of $\varphi_{y,x}$. See \cref{eq:ThomMap}. By suspending this map, we get a map
    \begin{equation*}
        g_{y,x}: S^{Nb}\smashSpaces (\Hlf^{J_{y,x}})_+ \to  Th(e_{y,x}) \to S^{Nb}.    
    \end{equation*}

    It's remind to show the functoriality of $K$, which amounts to
    \begin{equation*}
        \partial_y  g_{z,x} =  g_{y,x} \circ g_{z,y},
    \end{equation*}
    this equation holds due the compatibility of the neat embedding and the framing. See \cref{eq:ThomFunctoriality}.
\end{constr}

For a flow category $\fC$, we define $\fC^{[b,a]}$ to be the full subcategory generated by $c \in \fC$ such that $b \geq \norm{x} \geq a$. We denote by $K_{[b,a]}$ the topological chain complex of $\fC^{[b,a]}$. Note that there is natural transformation
\begin{equation*}
    \beta_{b+1}:K_{[b,a]} \to K_{[b+1,a]}
\end{equation*}
which induces from suspending further, to $N(b+1)$, lastly if $i=b+1$ we get that $\beta_{b+1}(i)$ is the zero map $0 \to S^{N(b+1)}$.

Similarly we have maps
\begin{equation*}
    \alpha_{a-1}:K_{[b,a-1]} \to K_{[b,a]}.
\end{equation*}

\begin{defn}
    We extend \cref{con:topChOfCat} to unbounded flow categories by taking (co)limit\todo{rewrite \cref{con:topChOfCat} as a claim about bounded flow cat}
    \begin{equation*}
        K_{\fC} = \lim_a \colim_b K^{[b,a]}
    \end{equation*}
\end{defn}


Let $Sp^{O}$ be the category of orthogonal spectra. This is a topologically enriched model category for the stable homotopy category, as developed in \cite{MMSS01}. It is equipped with a symmetric monoidal smash product $\otimes$ that gives a Quillen equivalent model to the $\infty$-category of spectra. In this model, the suspension spectrum functor is strongly symmetric monoidal:
\begin{equation*}
\Sigma^{\infty}: (Top_{+}, \wedge, S^{0}) \to (Sp^{O}, \otimes, \mathbb{S})
\end{equation*}
Furthermore, it preserves coproducts:
\begin{equation*}
\Sigma^{\infty}: (Top_{+}, \vee, *) \to (Sp^{O}, \oplus, 0)
\end{equation*}

\begin{constr}
    Let $\fC$ be an embedded framed flow category bounded by $b$ and $0$, as above. Using \cref{con:topChOfCat} and post composition with $\Sinf$ we construct a functor
    \begin{equation*}
        K_{\fC}' \in \Fun(J,\SPO),
    \end{equation*}
    with
    \begin{equation*}
        K_{\fC}'(i) = \WedgeSp_{\norm{x}=i} \bbS^{Nb}.
    \end{equation*}
    Using post composition with $\Sigma^{-Nb}: \Sp \to \Sp$ we define the chain complex of $\fC$ to be
    \begin{equation*}
        K_{\fC} \coloneqq \Sigma^{-Nb} K_{\fC}'
    \end{equation*}
    which have $\bbS$ in each level.
\end{constr}


\begin{prop}\label{prop:J_equivalence}% was {cor:thin_top_model}
There is an equivalence of pointed $\infty$-categories:
\begin{equation*}
    N(\Ch) \simeq N_{\Top}(\J),
\end{equation*}
where $\Ch$ is the indexing category of chain complexes, see \cref{def:Ch} and $N,N_{\Top}$ are the nerve and the topological nerve.
\end{prop}

\begin{proof}
The morphism spaces $\Hom_{\J}(j,i)$ are contractible when $j > i + 1$ (since $(\Hlf^{J_{j,i}})_+$ is contractible), and $\{ \id, \infty \} \simeq S^0$ when $i = j$, and $\{ d_i, \infty \} \simeq S^0$ when $j = i + 1$. 
The equivalence follows by considering both $\Ch$ and $\J$ as models for the same $\infty$-category.
\end{proof}

\begin{cor}
    In light of this equivalence we can start with framed flow category and obtain a coherent chain complex in spectra
    \begin{equation*}
    \Fun_{\Top}(\J,\SPO) \simeq \Fun(N_{\Top}(\J),N_{\Top}(\SPO)) \simeq \Fun(N(\Ch),\Sp) \eqqcolon \Ch(Sp)
\end{equation*}
\end{cor}

We will, by abuse of notation, identify $1$-category with it's nerves when viewing them as $\infty$-categories.

Let $\fC$ be a bounded, embedded, framed flow category and let $K \in \Fun(\Ch,\Sp)$ be its coherent chain complex. By \cref{sec:Ch} we get a spectral sequence $(\rnp{E},d_r)$. Note that it's immediate result that
\begin{equation*}
    \pg{E}{1}{n}{p} = \pi_{n-p} K_p = \pi_{n-p}\WedgeSpaces_{\norm{x}=i} \bbS\<{x}
\end{equation*}
where $\<{x}$ denote the $x$ component of the sum.
This is the analog of the Morse chain complex. However, the description of the higher pages and the differentials is obfuscated.

The first observation is to identify stable homotopy groups with framed bordism:
\begin{equation}
    \pi_{n-p} \bbS \simeq \Omega^{\fr}_{n-p}
\end{equation}
Let $y,x \in \fC$ be two objects with $\norm{y}=p, \norm{x}=p-1$. The fact that we used the Pontrjagin-Thom collapse map to define $K$ yields that the first differential 
\begin{equation*}
    d_1\colon \Omega^{\fr}_{n-p}\<{y} \to \Omega^{\fr}_{(n-1)-(p-1)}\<{x}
\end{equation*}
is given by
\begin{equation*}
    d_1(N) = N\times M_{y,x}
\end{equation*}
where $M_{y,x}$ is the morphism manifold between $y$ and $x$ in $\fC$ and $N$ is a representative for it's class in the bordism group. We get that $N$ survive to the second page if there is a $n-p+1$ dimensional manifold with boundary $W$, such that $\partial W = M_{y,x}\times N$ which is equivalent to the existence of a flow category with 3 objects, $s,y,x$ and morphism manifolds $M_{y,x},N=M_{s,y},W=M_{s,x}$.
