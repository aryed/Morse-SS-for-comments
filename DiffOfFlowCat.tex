\begin{defn}
    Let $b,a \in \bbZ$. Let $\fC$ and $\fD$ be framed flow categories bounded by $[b+1,a]$ and $[b,a-1]$ respectively. We say that $\fC$ and $\fD$ are \textbf{composable} if
    \begin{equation*}
        \fC_{[b,a]} = \fD_{[b,a]}
    \end{equation*}
    where $\fC_{[b,a]}$ is the full subcategory generated by $c \in \fC$ such that $b \geq \norm{x} \geq a$.

    Similarly two chain complexes $K,L$ (with same bounds) are composable if $K,L$ are composable as morphisms (see \cref{cor:ch_to_morph}).
\end{defn}

\begin{lem}\label{lem:colimIsMfld}
    Let $A$ be a finite set, $N \in \bbN$ and let 
    \begin{equation*}
        D: \op^A \setminus \const{0} \to \Top
    \end{equation*}
    be a diagram such that for each $a \in A$ the restriction
    \begin{equation*}
        \op^{A\setminus \{a\}} \to \op^{A} \to \Top
    \end{equation*}
    is $\norm{A}-1$ dimensional $\<{A \setminus\{a\}}$ manifold. Then the topological space
    \begin{equation*}
        X \coloneqq \colim_{\{1\to 0\}^A \setminus \const{0}} D
    \end{equation*}
    inherits a structure of a closed $\norm{A}-1$ manifold. Moreover, if $D(a)$ are framed, then $X$ inherits a framing.
\end{lem}
We refer to such diagrams as "punctured cubes".

\begin{proof}
    The idea is to use the following homeomorphism:
    \begin{equation*}
        \Hlf^2 \simeq \Hlf \times \bbR
    \end{equation*}
    which, in polar coordinates, is given by 
    \begin{equation*}
        (r,\theta) \mapsto (r,2\theta).
    \end{equation*}
    Inductively we get that for any finite set $A$ and $a \in A$ we have
    \begin{equation*}
        \Hlf^A \simeq H \times \bbR^{A\setminus\{a\}}
    \end{equation*}
    which induces the homeomorphism on the boundary
    \begin{equation}\label{eq:flat}
        \partial \Hlf^A \simeq \bbR^{A\setminus\{a\}},
    \end{equation}
    see \cref{def:BoundaryOfADiag}.

    The fact that the restrictions of $D$ are $\<{A\setminus\{a\}}$ manifolds give them the local model of 
    \begin{equation*}
        \Hlf^{A\setminus\{a\}}
    \end{equation*}
    hence, for $B \subset A, a \in B$, locally $D(B)$ is given by
    \begin{equation*}
         \Hlf^{A\setminus\{a\}}(B)  = \partial_a \Hlf^A(B)
    \end{equation*}
     where $\Hlf^{A}$ is the diagram of the manifolds with corners see \cref{ex:mfld_diagram} and we view $\partial_a \Hlf^A$ as a sub-diagram of $\Hlf^A$. Under this identification we have that
    \begin{equation*}
        \partial_a \Hlf^A(B) = \Hlf^A(B)
    \end{equation*}
    where $D(B)$ is locally homeomorphic to $\Hlf^A(B)$.
    
    By taking colimit:
    \begin{equation*}
        \colim_{\{1\to 0\}^A \setminus \const{0}}  \Hlf^{A} = \partial (\Hlf^{A}) 
    \end{equation*}
    we get that any point of $X$ have an neighborhood homeomorphic to an open set of $\partial \Hlf^{A}$, to get a manifold structure on $X$ we use \cref{eq:flat} and get that $X$ is locally given by 
    \begin{equation*}
        \partial (\Hlf^{A}) \simeq \bbR^{A\setminus\{a\}} \simeq \bbR^{\norm{A}-1}.
    \end{equation*}

    Assume further that $D(a)$ are embedded and normally framed manifolds with framing $\varphi_a$.  The natural transformations $\varphi_a$

    \end{equation*}
    where the left is the diagram $\nu D(B)$ and the right is $D(B) \times \bbR^{N}$. By taking colimit we get a normal framing
    \begin{equation*}
        \nu X \simeq X \times \bbR^{N}.
    \end{equation*}

\begin{exmp}
    Take $\fC$ to be a frmaed flow category with two objects with indices $1,0$ and let $W,V$ be right and left bimodules over $\fC$ with degrees $1,0$ respectively. The data of $\fC$ is a manifold $M_{1,0}$, and $W,V$ amounts to a closed $0$-dimensional manifolds $W_1, V_0$ and frmaed $1$-dimensional manifolds $W_0, V_1$ with boundary
    \begin{equation*}
        \partial W_0 \simeq W_1 \times M_{1,0}, \quad \partial V_1 \simeq M_{1,0} \times V_0.
    \end{equation*}
    The construction is given by gluing
    \begin{equation*}
        W \circ V \simeq W_0 \times V_1 \cup V_0 \times W_1
    \end{equation*}
    along $W_1\times M_{1,0} \times V_0$.
\end{exmp}

\begin{defn}\label{def:CompCat}
    Let $W,V$ be right (left) modules over a flow category $\fC$ with degree $i$ (resp $j$), with flow manifolds $W_{-},M_{-,-},V_{-}$. We define
    \begin{equation*}
        D \colon \op^{\{i,\dots, j\}} \setminus \const{0} \to \Top
    \end{equation*}
    in the following way: for each $B = \{b_k,\dots b_1\} \subseteq \{i,\dots, j\}$
    \begin{equation*}
        D(b_k,\dots b_1) = W_{b_k} \times M_{b_k,b_{k-1}} \times \dots \times M_{b_2,b_1} \times V_{b_1}.
    \end{equation*}
    We write
    \begin{equation*}   
        W \circ V \coloneqq \bigcup_{i \leq p \leq j} W_{p} \times V_{p} \coloneqq \colim D.
    \end{equation*}
\end{defn}
Note that $W \circ V: *[i] \to *[j]$, hence consist of only one manifold, so this notation consistent with the flow modules notation of the paper.

\begin{claim}\label{claim:compColim}
    Let $D$ be a punctured diagram as in \cref{lem:colimIsMfld}, then we have that
    \begin{equation*}
        (\colim D(B))_+ \simeq \colim (D(B)_+)
    \end{equation*}
\end{claim}
\begin{proof}
    Note that 
    \begin{equation*}
        \colim D(B)_+ = \colim (D(B) \cup \{\infty_B\}) \simeq (\colim D(B)) \cup \{\infty_A\}
    \end{equation*}
    where that last equivalence is the identification of all the points with the point of $D(A)$. This resulting a compact space, which is differ from $\colim D$ by one point, hence we get that 
    \begin{equation*}
        \colim D(B)_+ \simeq (\colim D)_+
    \end{equation*}
\end{proof}

\begin{lem}
    Let $\fC,\fD$ be two composable framed flow categories, then We have that
    \begin{equation*}
        K_{\fD \circ \fC} \simeq K_{\fD} \circ K_{\fC}
    \end{equation*}
\end{lem}
\begin{proof}
    First we remind that $K_{\fC}$ is the chain complexes with maps
    \begin{equation*}
        f_{z,y}: \Hlf^{J_{z,y}} \smashSpaces S^{N(b+1-a)} \to S^{N(b+1-a)}
    \end{equation*}
    which constructed by collapsing $M_{z,y}$. Similarly the chain complex $K_{\fD}$ have the maps $g_{y,x}$ which constructed by collapsing $N_{y,x}$.
    
    Using the topologically enriched model, by \cref{lem:diff_is_comp} our goal is to show that
    \begin{equation*}
        c(\bigcup_{b \geq p\geq a} M_{b+1,p}\times N_{p,a-1}) = \bigcup_{b \geq p\geq a} f_{b+1,p}\smashSpaces g_{p,a-1}
    \end{equation*}
    where $c$ is the Pontrjagin-Thom collapse map.

    Let $D$ be a punctured cube diagram as above. Let $X \coloneqq \colim D$ be the colimit manifold.
    We denote
    \begin{equation*}
        A = J_{b+1,a-1} = \{b,\dots a\}
    \end{equation*}
    and note that $\norm{A} = b-a+1$.
    We remind that the Pontryagin Thom map of $X$ is constructed in the following way: for each $B \subseteq \{b,\dots a\}$ we have a neat embedding
    \begin{equation*}
        e(B) \colon D(B) \hookrightarrow \Hlf^A(B) \times \bbR^{N(b-a+2)}
    \end{equation*}
    by taking colimit we get the assembly map
    \begin{equation*}
        X = \colim_{\{1\to 0\}^{A} \setminus \const{0}} D \hookrightarrow
        \colim_{\{1\to 0\}^{A} \setminus \const{0}} \Hlf^{A} \times \bbR^{N(b-a+2)} \simeq \bbR^{\norm{A}-1} \times \bbR^{N(b-a+2)}.
    \end{equation*}
    We get an open embedding of the normal bundle
    \begin{equation*}
        e \colon \nu X \hookrightarrow \bbR^{\norm{A}-1} \times \bbR^{N(b-a+2)}
    \end{equation*}
    which is the colimit of the maps
    \begin{equation*}
        e(B) \colon \nu D(B) \hookrightarrow \Hlf^A(B) \times \bbR^{N(b-a+2)}.
    \end{equation*}
    On the other hand we have the normal framing, compost with the projection
    \begin{equation*}
        \varphi(B) \colon \nu D(B) \simeq D(B) \times \bbR^{N(b-a+2)} \to \bbR^{N(b-a+2)}
    \end{equation*}
    which assemble into the map
    \begin{equation*}
        \colim \varphi \colon \nu X \to \bbR^{N(b-a+2)}
    \end{equation*}
    The Thom map (see \cref{eq:ThomMap}) of $X$ is given by
    \begin{equation*}
        c_X = \varphi_+ \circ e_+
    \end{equation*}
    where $e_+$ is given by collapsing the complement of $\nu X$.
    We can re-write this map as
    \begin{equation*}
        (\colim \varphi(B))_+ \circ \colim(e(B))_+
    \end{equation*}
    and using \cref{claim:compColim} we get that
    \begin{equation*}
        (\colim \varphi(B)_+) \circ \colim(e(B)_+) = \colim(\varphi(B)_+ \circ  e(B)_+)
    \end{equation*}
    which is the colimit over the diagram of taking point-wise Thom maps of $D(B)$. To finish the proof we need to show that the Thom maps of 
    \begin{equation*}
        D(\{p\}) = M_{b+1,p} \times N_{p,a-1}
    \end{equation*}
    are homotopic to $f_{b+1,p}\smashSpaces g_{p,a-1}$.
    This follows from the functoriality of $(-)_+$ and by isotoping the tubular neighborhoods
    \begin{equation*}
        \nu (M_{b+1,p} \times N_{p,a-1})  \text{ and } \nu M_{b+1,p} \times \nu N_{p,a-1}.
    \end{equation*}
\end{proof}

\begin{thm}\label{cor:SSofFFCat}
    Let $\fC$ be a flow category with a neat embedding and normal framing. Let $K_{\fC}$ be the chain complex obtained by collapsing the morphism manifolds of $\fC$, and denote by $\rnp{E}$ the spectral sequence of $K_{\fC}$. We have that:
    \begin{equation*}
        \pg{E}{1}{n}{p}=\bigoplus_{\norm{x}=p} \Omega^{\fr}_{n-p}.
    \end{equation*}
    A collection of closed framed manifolds $L = \{L_z\}_{\norm{z}=p}$ representing an element in $\pg{E}{1}{n}{p}$ survives to the $(r+1)$-th page if and only if $L$ extends to a right flow module $W$ of degree $n$ over $\fC_{[p, p-r]}$ (see \cref{def:ModuleSurvives}).
    
    If $L$ survives to the $(r+1)$-th page, then the differential $d_{r+1}(L)$ is given by the bordism class of the Toda manifold $W \circ \flowM_{-, p-r-1}$ (the geometric tensor product of the right module $W$ and the left module $\flowM_{-, p-r-1}$), which is represented by the closed framed manifold:
    \begin{equation*}
        \bigcup_{p \geq q \geq p-r} W_q \times M_{q, p-r-1}
    \end{equation*}
    where the $W_q$ are the manifolds in the flow module $W$, and the $M_{q, p-r-1}$ are the morphism manifolds of $\fC$.
\end{thm}
\begin{proof}
    On the one hand, assume that the collection $L$ extends to a right flow module $W$ over $\fC_{[p,p-r]}$ of degree $n$. Using \cref{con:topChOfCat}, we construct a map of coherent chain complexes (or a $\J$-module map) corresponding to $W$:
    \begin{equation*}
        K_W: K_{S_n} \to K_{\fC_{[p,p-r]}}
    \end{equation*}
    which extends the map representing $L$ on the degree $p$ objects. By the homological criterion established in \cref{lem:ChSur}, this map of chain complexes guarantees that the element $L$ survives to the $(r+1)$-th page of the spectral sequence.
    
    For the differential, let $W$ be the right flow module of degree $n$ over $\fC_{[p,p-r]}$ extending $L$. By \cref{lem:Ch_diff}, the differential $d_{r+1}(L)$ is given by the composition of the chain maps:
    \begin{equation*}
        K_{\flowM_{-, p-r-1}} \circ K_W: K_{S_n} \to K_{p-r-1}
    \end{equation*}
    where the first map is the left flow module $\flowM_{-, p-r-1}$ viewed as a chain map.
    Under the equivalence between compositions of chain complexes and compositions of modules, this corresponds to the Toda manifold $W \circ \flowM_{-, p-r-1}$, which is represented by the closed framed manifold:
    \begin{equation*}
        \bigcup_{p \geq q \geq p-r} W_q \times M_{q, p-r-1}
    \end{equation*}
    where the $W_q$ are the manifolds of the module $W$ and $M_{q, p-r-1}$ are the morphism manifolds of $\fC$.
    
    The remaining part is to show that if $L \in \pg{E}{1}{n}{p}$ survives to the $(r+2)$-th page, then the collection $L$ extends to a right flow module over $\fC_{[p, p-r-1]}$. By the definition of survival, if $L$ survives then $d_{r+1}(L) = 0$ in bordism. Thus, the Toda manifold $W \circ \flowM_{-, p-r-1}$ is null-bordant, meaning there exists a framed manifold with corners $W_{p-r-1}$ of dimension $n-p+r$ such that:
    \begin{equation*}
        \partial W_{p-r-1} = W \circ \flowM_{-, p-r-1} = \bigcup_{p \geq q\geq p-r} W_q\times M_{q,p-r-1}.
    \end{equation*}
    By adding the manifold $W_{p-r-1}$ to the collection $\{W_q\}_{q \in \fC_{[p,p-r]}}$, we obtain a right flow module over $\fC_{[p, p-r-1]}$ of degree $n$, as required.

\end{proof}
