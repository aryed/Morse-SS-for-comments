\begin{notation}
    For two framed flow categories $\fC,\fD$ with flow manifolds $M_{-,-},N_{-,-}$ we denote by
    \begin{equation*}
        \fC = \fD
    \end{equation*}
    the following relation: there is a bijection 
    \begin{align*}
        i \colon \Ob(\fC) &\simeq \Ob(\fD)\\
        M_{x,y} &= N_{i(x),i(y)}
    \end{align*}
\end{notation}

\begin{defn}
    Let $b,a \in \bbZ$ and $\fC, \fD$ be a framed flow categories which bounded by $b+1,a$ and $b,a-1$ (resp.).\todo{looks ugly} We say that $\fC,\fD$ are \textbf{composable} if
    \begin{equation*}
        \fC_{[b,a]} = \fD_{[b,a]}
    \end{equation*}
    where $\fC_{[b,a]}$ is the full subcategory generated by $c \in \fC$ such that $b \geq \norm{x} \geq a$.

    Similarly two chain complexes $K,L$ (with same bounds) are composable if $K,L$ are composable as morphisms (see \cref{cor:ch_to_morph}).
\end{defn}

Note that the target of $K$ is $K_{[b,a]}$ and the source of $L$ is $L_{[b,a]}$ we deduce that for any composable categories $\fC,\fD$, $K_{\fC},K_{\fD}$ are composable chain complexes.

Our goal is to define $\fD \circ \fC$ to be a flow category such that
\begin{equation*}
    K_{\fD \circ \fC} \simeq K_{\fD} \circ K_{\fC}.
\end{equation*}

Denote by $M_{y,x}, N_{y,x}$ the morphism manifolds of $\fC,\fD$, the idea is to define $\fD \circ \fC$ to be a flow category with objects wit indices $b+1,a-1$ and one morphism manifold
\begin{equation*}
    \bigcup_{b \geq p\geq a} M_{b+1,p}\times N_{p,a-1}
\end{equation*}
We begin by justifying the structure of this union of manifolds.

\begin{claim}
    Let $\fC, \fD$ be two composable flow categories indexed on $b+1,a$ and $b,a-1$. Denote by $M_{y,x}$ and $N_{y,x}$ their morphism manifolds, Then 
    \begin{equation*}
        \bigcup_{b \geq p\geq a} M_{b+1,p}\times N_{p,a-1}
    \end{equation*}
    is a closed framed manifold with dimension $b-a+1$.
\end{claim}
The above claim will follow from the general following "flattening" argument.
\begin{lem}\label{lem:colimIsMfld}
    Let $A$ be a finite set, $N \in \bbN$ and let 
    \begin{equation*}
        D: \op^A \setminus \const{0} \to \Top
    \end{equation*}
    be a diagram such that for each $a \in A$ the restriction
    \begin{equation*}
        \op^{A\setminus \{a\}} \to \op^{A} \to \Top
    \end{equation*}
    is $\norm{A}-1$ dimensional $\<{A \setminus\{a\}}$ manifold. Then the topological space
    \begin{equation*}
        X \coloneqq \colim_{\{1\to 0\}^A \setminus \const{0}} D
    \end{equation*}
    inherits a structure of a closed $\norm{A}-1$ manifold. Moreover, if $D(a)$ are framed, then $X$ inherits a framing.
\end{lem}
We refer to such diagrams as "punctured cubes".

\begin{proof}
    The idea is to use the following homeomorphism:
    \begin{equation*}
        \Hlf^2 \simeq \Hlf \times \bbR
    \end{equation*}
    which, in polar coordinates, is given by 
    \begin{equation*}
        (r,\theta) \mapsto (r,2\theta).
    \end{equation*}
    Inductively we get that for any finite set $A$ and $a \in A$ we have
    \begin{equation*}
        \Hlf^A \simeq H \times \bbR^{A\setminus\{a\}}
    \end{equation*}
    which induces the homeomorphism on the boundary
    \begin{equation}\label{eq:flat}
        \partial \Hlf^A \simeq \bbR^{A\setminus\{a\}},
    \end{equation}
    see \cref{def:BoundaryOfADiag}.

    The fact that the restrictions of $D$ are $\<{A\setminus\{a\}}$ manifolds give them the local model of 
    \begin{equation*}
        \Hlf^{A\setminus\{a\}}
    \end{equation*}
    hence, for $B \subset A, a \in B$, locally $D(B)$ is given by
    \begin{equation*}
         \Hlf^{A\setminus\{a\}}(B)  = \partial_a \Hlf^A(B)
    \end{equation*}
     where $\Hlf^{A}$ is the diagram of the manifolds with corners see \cref{ex:mfld_diagram} and we view $\partial_a \Hlf^A$ as a sub-diagram of $\Hlf^A$. Under this identification we have that
    \begin{equation*}
        \partial_a \Hlf^A(B) = \Hlf^A(B)
    \end{equation*}
    where $D(B)$ is locally homeomorphic to $\Hlf^A(B)$.
    
    By taking colimit:
    \begin{equation*}
        \colim_{\{1\to 0\}^A \setminus \const{0}}  \Hlf^{A} = \partial (\Hlf^{A}) 
    \end{equation*}
    we get that any point of $X$ have an neighborhood homeomorphic to an open set of $\partial \Hlf^{A}$, to get a manifold structure on $X$ we use \cref{eq:flat} and get that $X$ is locally given by 
    \begin{equation*}
        \partial (\Hlf^{A}) \simeq \bbR^{A\setminus\{a\}} \simeq \bbR^{\norm{A}-1}.
    \end{equation*}

    Assume further that $D(a)$ are embedded and normally framed manifolds with framing $\varphi_a$.  The natural transformations $\varphi_a$
    \begin{equation*}
        \nu D(a) \simeq D(a) \times \bbR^N
    \end{equation*}
    (where $N \in \bbN$) assemble into a neutral transformation
    \begin{equation*}
        \nu D \simeq D \times \bbR^N
    \end{equation*}
    where the left is the diagram $\nu D(B)$ and the right is $D(B) \times \bbR^{N}$. By taking colimit we get a normal framing
    \begin{equation*}
        \nu X \simeq X \times \bbR^{N}.
    \end{equation*}

\begin{exmp}
    Take $b=2,a=1$ and take $\fC,\fD$ to be flow categories with tree objects. The data of $\fC$ amounts to two closed $0$-dimensional manifolds $M_{3,2}, M_{2,1}$ and a $1$-dimensional manifold $M_{3,1}$ with boundary
    \begin{equation*}
        \partial M_{3,1} = M_{3,2} \times M_{2,1}
    \end{equation*}
    and similar for $\fD$ with manifolds $N_{2,1}, N_{1,0}$ and $N_{2,0}$, where $M_{2,1}=N_{2,1}$.
    The construction is given by gluing
    \begin{equation*}
        M_{3,1}\times N_{1,0} \cup M_{3,2}\times N_{2,0}
    \end{equation*}
    along $M_{3,2}\times M_{2,1} \times N_{1,0}$.
\end{exmp}
This example recovers the secondary structure on bordism rings, introduced by Alexander in \cite{alexander1972cobordism}. We can think of $\fD \circ \fC$ as the \emph{Toda brackets} or \emph{Massey product} of the $0$-dimensional manifolds of $\fC,\fD$.

\begin{defn}\label{def:CompCat}
    Let $\fC,\fD$ be two composable flow categories with flow manifolds $M_{-,-},N_{-,-}$. We define
    \begin{equation*}
        D \colon \op^{\{b,\dots a\}} \setminus \const{0} \to \Top
    \end{equation*}
    in the following way: for each $B = \{i_k,\dots i_1\} \subseteq \{b,\dots a\}$
    \begin{equation*}
        D(i_k,\dots i_1) = M_{b+1,i_k}M_{i_k,i_{k-1}} \times \dots M_{i_2,i_1} \times N_{i_1,a-1}.
    \end{equation*}
    We write
    \begin{equation*}
        M_{b+1,a-1} \coloneqq \bigcup_{b \leq p \leq a} M_{b+1,p} \times N_{p,a-1} \coloneqq \colim D.
    \end{equation*}
    The \textbf{composition} flow category $\fC \circ \fD$, is given by
    \begin{equation*}
        \ob(\fD \circ \fC) \coloneqq \{y,x\mid y \in \fC_{b+1}, x \in \fD_{a-1} \}, \quad \norm{y}_{\fD \circ \fC}=\norm{y}_{\fC}-1, \norm{x}_{\fD \circ \fC}=\norm{x}_{\fD}
    \end{equation*}
    and $M_{b+1,a-1}$ is the morphism manifold of this category.
\end{defn}

\begin{rem}
    In our convention $M_{b+1,a-1}$ is the union of all the morphism manifolds of $\fC \circ \fD$. To get the morphism manifolds $M_{y,x}$ between two objects $y,x \in \fC \circ \fD$, one need to modify the colimit by replacing $M_{b+1,b_k}$ with $M_{y,b_k}$ and $M_{b_1,a-1}$ with $M_{b_1,x}$.
\end{rem}

\begin{claim}\label{claim:compColim}
    Let $D$ be a punctured diagram as in \cref{lem:colimIsMfld}, then we have that
    \begin{equation*}
        (\colim D(B))_+ \simeq \colim (D(B)_+)
    \end{equation*}
\end{claim}
\begin{proof}
    Note that 
    \begin{equation*}
        \colim D(B)_+ = \colim (D(B) \cup \{\infty_B\}) \simeq (\colim D(B)) \cup \{\infty_A\}
    \end{equation*}
    where that last equivalence is the identification of all the points with the point of $D(A)$. This resulting a compact space, which is differ from $\colim D$ by one point, hence we get that 
    \begin{equation*}
        \colim D(B)_+ \simeq (\colim D)_+
    \end{equation*}
\end{proof}

\begin{lem}
    Let $\fC,\fD$ be two composable framed flow categories, then We have that
    \begin{equation*}
        K_{\fD \circ \fC} \simeq K_{\fD} \circ K_{\fC}
    \end{equation*}
\end{lem}
\begin{proof}
    First we remind that $K_{\fC}$ is the chain complexes with maps
    \begin{equation*}
        f_{z,y}: \Hlf^{J_{z,y}} \smashSpaces S^{N(b+1-a)} \to S^{N(b+1-a)}
    \end{equation*}
    which constructed by collapsing $M_{z,y}$. Similarly the chain complex $K_{\fD}$ have the maps $g_{y,x}$ which constructed by collapsing $N_{y,x}$.
    
    Using the topologically enriched model, by \cref{lem:diff_is_comp} our goal is to show that
    \begin{equation*}
        c(\bigcup_{b \geq p\geq a} M_{b+1,p}\times N_{p,a-1}) = \bigcup_{b \geq p\geq a} f_{b+1,p}\smashSpaces g_{p,a-1}
    \end{equation*}
    where $c$ is the Pontrjagin-Thom collapse map.

    Let $D$ be a punctured cube diagram as above. Let $X \coloneqq \colim D$ be the colimit manifold.
    We denote
    \begin{equation*}
        A = J_{b+1,a-1} = \{b,\dots a\}
    \end{equation*}
    and note that $\norm{A} = b-a+1$.
    We remind that the Pontryagin Thom map of $X$ is constructed in the following way: for each $B \subseteq \{b,\dots a\}$ we have a neat embedding
    \begin{equation*}
        e(B) \colon D(B) \hookrightarrow \Hlf^A(B) \times \bbR^{N(b-a+2)}
    \end{equation*}
    by taking colimit we get the assembly map
    \begin{equation*}
        X = \colim_{\{1\to 0\}^{A} \setminus \const{0}} D \hookrightarrow
        \colim_{\{1\to 0\}^{A} \setminus \const{0}} \Hlf^{A} \times \bbR^{N(b-a+2)} \simeq \bbR^{\norm{A}-1} \times \bbR^{N(b-a+2)}.
    \end{equation*}
    We get an open embedding of the normal 
    \begin{equation*}
        e \colon \nu X \hookrightarrow \bbR^{\norm{A}-1} \times \bbR^{N(b-a+2)}
    \end{equation*}
    which is the colimit of the maps
    \begin{equation*}
        e(B) \colon \nu D(B) \hookrightarrow \Hlf^A(B) \times \bbR^{N(b-a+2)}.
    \end{equation*}
    On the other hand we have the normal framing, compost with the projection
    \begin{equation*}
        \varphi(B) \colon \nu D(B) \simeq D(B) \times \bbR^{N(b-a+2)} \to \bbR^{N(b-a+2)}
    \end{equation*}
    which assemble into the map
    \begin{equation*}
        \colim \varphi \colon \nu X \to \bbR^{N(b-a+2)}
    \end{equation*}
    The Thom map (see \cref{eq:ThomMap}) of $X$ is given by
    \begin{equation*}
        c_X = \varphi_+ \circ e_+
    \end{equation*}
    where $e_+$ is given by collapsing the complement of $\nu X$.
    We can re-write this map as
    \begin{equation*}
        (\colim \varphi(B))_+ \circ \colim(e(B))_+
    \end{equation*}
    and using \cref{claim:compColim} we get that
    \begin{equation*}
        (\colim \varphi(B)_+) \circ \colim(e(B)_+) = \colim(\varphi(B)_+ \circ  e(B)_+)
    \end{equation*}
    which is the colimit over the diagram of taking point-wise Thom maps of $D(B)$. To finish the proof we need to show that the Thom maps of 
    \begin{equation*}
        D(\{p\}) = M_{b+1,p} \times N_{p,a-1}
    \end{equation*}
    are homotopic to $f_{b+1,p}\smashSpaces g_{p,a-1}$.
    This follows from the functoriality of $(-)_+$ and by isotoping the tubular neighborhoods
    \begin{equation*}
        \nu (M_{b+1,p} \times N_{p,a-1})  \text{ and } \nu M_{b+1,p} \times \nu N_{p,a-1}.
    \end{equation*}
\end{proof}
