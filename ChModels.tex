\subsection{Flattening $n+1$ cubes}\label{sec:ChModels.Cub}

\begin{defn}\label{def:A_cube}
Let $A$ be a finite linearly ordered set, the \emph{$A$-cube} is the power set category:
\[
    \op^A \cong \mathcal{P}(A)^{\Op}
\]
For $A = \{1, \dots, n\}$, we write $\op^n = \op^A$.
\end{defn}

\begin{note*}
For brevity we will use $[1]$ along this section, and at the end we will "opposite" it so our cubes (and chains) will be decreasing and not increasing.
\end{note*}

\begin{defn}\label{def:spine_of_A}
Let $A$ be a finite linearly ordered set, let $a \in A$, define $s_a \subseteq A$ to be the prefix subset
\[
    s_a \coloneq \{ b \in A \mid b \leq a \}.
\]
The spine of $A$ is the full subcategory of $[1]^A$ spanned by 
\[
     \Spine(A) \coloneq \{ \varnothing \} \cup \{ s_a \mid a \in A \}
\]
\end{defn}

\begin{defn}\label{def:category_of_spine_cubes}
Let $\cC$ be a pointed $\infty$-category and let $A$ be a finite linearly ordered set. We define the category of spine cubes (or cubical chain complexes)
\[
    \Cu_A(\cC) :=  \Fun_*^{\Spine}([1]^A,\cC) \coloneq \{ F \in \Fun_*([1]^A,\cC) \mid F(Q) \simeq * \text{ for all } Q \notin \Spine(A) \}
\]
\end{defn}

Let $A,B$ be finite linearly ordered sets, then
\[
    [1]^{A \amalg B} \simeq [1]^A \times [1]^B,
\]
hence
\[
    \Fun_*([1]^{A \amalg B},\cC) \simeq \Fun_*([1]^A \times [1]^B,\cC) \simeq \Fun_*([1]^A, \Fun_*([1]^B, \cC)).
\]

By taking $A=\{a\}$, we obtain
\[
    \Fun_*([1]^{\{a\} \amalg B},\cC) \simeq \Fun_*([1]^{\{a\}} \times [1]^B,\cC) \simeq \Fun_*([1], \Fun_*([1]^B, \cC)).
\]

We think of this equivalence as "flattening" the morphism of cubes (RHS) into a longer cube (the LHS).

Passing to the spine cubes we need to be careful:
Let $B=\{b_1 < \dots < b_n\}$ be a linearly ordered set, and let $a$ be an element smaller than all elements of $B$. We have that the spine of $\{a\} \amalg B$ inside $[1]^{\{a\} \amalg B} \simeq [1] \times [1]^B$ consists of the elements:
\[
    (0, 0, \dots, 0), \quad (1, 0, \dots, 0), \quad (1, 1, 0, \dots, 0), \quad \dots, \quad (1, 1, \dots, 1)
\]
which can be written as:
\[
    \{ (0, \varnothing) \} \cup \left( \{1\} \times \Spine(B) \right) \subset [1] \times [1]^B.
\]
We get that for the cubical chain complexes we have:
\begin{align}
    \Cu_{\{a\} \amalg B}(\cC) &\simeq \Fun_*^{\Spine}([1]^{\{a\} \amalg B},\cC) \\
    &\simeq \big\{f \colon [1] \to \Fun_*^{\Spine}([1]^B, \cC) \mid
    \begin{aligned}
        f_0(Q) &\simeq * \quad \text{for all } \varnothing \neq Q \subseteq B, \\    
        f_1(Q) &\simeq * \quad \text{for all } Q \notin \Spine(B)
    \end{aligned}
     \big\}
\end{align}

where $f_0,f_1$ are the source and the target of the morphism $f$.

We phrase it as follows:
\begin{lem}\label{lem:morphism_to_cube}
Let $L, M \in \Cu_n(\cC)$ be two cubical chain complexes such that $L$ is concentrated at the initial object (i.e., $L(Q) \simeq *$ for all $Q \neq \varnothing$). Given a morphism $f \colon L \to M$, there is a corresponding cubical chain complex $K_f \in \Cu_{n+1}(\cC)$ whose source is $L$ and target is $M$. 
Conversely, given $K \in \Cu_{n+1}(\cC)$, there is a corresponding morphism $f_K \colon K|_{\{0\} \times [1]^n} \to K|_{\{1\} \times [1]^n}$ in $\Cu_n(\cC)$, where $K|_{\{0\} \times [1]^n}$ is concentrated at the initial object.
\end{lem}
